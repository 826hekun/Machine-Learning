# 这段代码使用一个循环来遍历参数组合。在每次迭代中，它调用ranger()函数来训练一个随机森林模型，并记录运行时间和RMSE。ranger()函数的参数如下：
# 
# formula：指定响应变量和预测变量的公式。
# data：包含响应变量和预测变量的数据框。
# num.trees：决策树的数量。
# mtry：决策树在每个分裂中要考虑的预测变量数量。
# min.node.size：决策树中允许的最小叶子节点大小。
# respect.unordered.factors：指定随机森林如何处理无序因子变量。
# verbose：设置是否输出详细信息。
# seed：设置随机种子，以确保结果的可重复性。
# 最后，代码使用ggplot2包绘制了“min.node.size”与RMSE之间的关系图，以及相同图中的“min.node.size”与运行时间之间的关系。
tuning_grid <- expand.grid(
  min.node.size = 1:20,
  run_time  = NA,
  rmse = NA
)

for(i in seq_len(nrow(tuning_grid))) {
  fit_time <- system.time({
    fit <- ranger(
    formula    = Sale_Price ~ ., 
    data       = ames_train, 
    num.trees  = 1000,
    mtry       = 26,
    min.node.size = tuning_grid$min.node.size[i],
    respect.unordered.factors = 'order',
    verbose    = FALSE,
    seed       = 123
  )
})
  
  tuning_grid$run_time[i] <- fit_time[[3]]
  tuning_grid$rmse[i] <- sqrt(fit$prediction.error)
  
}
# tuning_grid: 包含三列的数据框，分别为"min.node.size"（最小节点大小）、"run_time"（运行时间）和"rmse"（均方根误差），用于保存模型调参后的结果。
# 
# fit: ranger()函数拟合的模型对象，其中的参数有：
#

# data: 用于拟合模型的数据集。
# num.trees: 森林中的树的数量。
# mtry: 每个节点可选用的特征数。这里设置为26。

# 这是一个用于超参数调整的网格搜索。在每个循环迭代中，使用不同的参数组合运行随机森林模型，然后记录每个组合的误差（RMSE）。
# 
# 参数解释如下：
# 
# sample.fraction：用于构建每个树的训练数据的观测样本的分数。该参数的值通常在0.5和1之间。
# replace：指示观测样本是否应该替换。如果为TRUE，则观测样本可能被重复采样，否则每个样本只能用一次。
# rmse：每个参数组合的记录的均方根误差。
# 在循环的每次迭代中，使用expand.grid()函数生成参数网格，并使用seq_len(nrow(tuning_grid))函数迭代每个组合的参数。对于每个参数组合，运行ranger()函数来拟合随机森林模型，并使用sqrt(fit$prediction.error)函数计算预测误差的均方根。
# 
# 最终，tuning_grid包含所有测试参数组合的均方根误差值。
tuning_grid <- expand.grid(
  sample.fraction = seq(.05, .95, by = .05),
  replace  = c(TRUE, FALSE),
  rmse = NA
)

for(i in seq_len(nrow(tuning_grid))) {
  fit <- ranger(
    formula    = Sale_Price ~ ., 
    data       = ames_train, 
    num.trees  = 1000,
    mtry       = 26,
    sample.fraction = tuning_grid$sample.fraction[i],
    replace = tuning_grid$replace[i],
    respect.unordered.factors = 'order',
    verbose    = FALSE,
    seed       = 123
  )

  tuning_grid$rmse[i] <- sqrt(fit$prediction.error)
  
}
# 这段代码是针对随机森林中的采样方法进行参数调整的，其中涉及的参数包括：sample.fraction和replace。
# 
# sample.fraction表示样本被抽样的比例，可以看到这里设置的是从0.05到0.95每隔0.05一个间隔进行抽样。
# 
# replace表示样本是否有放回抽样，当replace=TRUE时，表示样本有放回抽样，当replace=FALSE时，表示样本无放回抽样。
# 
# 针对不同的参数组合，通过循环计算得到模型的预测误差（rmse），最终生成一个dataframe。
# 
# 最后使用ggplot进行可视化，其中x轴表示样本被抽样的比例，y轴表示模型的预测误差，点的颜色表示是否有放回抽样。可以通过观察不同参数组合下的误差和抽样比例的关系，来确定最优的参数组合。

# 这段代码创建了一个超参数网格，其中包含了4个超参数的所有组合，然后使用ranger函数对每个超参数组合进行模型拟合，最后将模型的评估指标OOB误差（rmse）存储在超参数网格中。
# 
# 具体而言，超参数网格包含以下超参数：
# 
# mtry：控制每个树的分裂所使用的特征数，由floor(n_features * c(.05, .15, .25, .333, .4))生成，即特征数n_features的5个不同百分比；
# min.node.size：控制每个叶节点最小样本数，由c(1, 3, 5, 10)生成，即4个不同的值；
# replace：一个布尔值，指示样本是否可以重复抽样，由c(TRUE, FALSE)生成，即两个不同的取值；
# sample.fraction：每次迭代时从训练集中抽样的样本比例，由c(.5, .63, .8)生成，即三个不同的取值。
# 在每个超参数组合中，ranger函数会拟合一个包含n_features * 10棵树的随机森林模型，并使用Out-of-Bag（OOB）误差来评估模型的性能，即在每次迭代中用未被抽取的样本集来评估模型的表现。然后，将得到的OOB误差（rmse）存储在超参数网格中，以便之后进行分析。
# create hyperparameter grid
hyper_grid <- expand.grid(
  mtry = floor(n_features * c(.05, .15, .25, .333, .4)),
  min.node.size = c(1, 3, 5, 10), 
  replace = c(TRUE, FALSE),                               
  sample.fraction = c(.5, .63, .8),                       
  rmse = NA                                               
)

# execute full cartesian grid search
for(i in seq_len(nrow(hyper_grid))) {
  # fit model for ith hyperparameter combination
  fit <- ranger(
    formula         = Sale_Price ~ ., 
    data            = ames_train, 
    num.trees       = n_features * 10,
    mtry            = hyper_grid$mtry[i],
    min.node.size   = hyper_grid$min.node.size[i],
    replace         = hyper_grid$replace[i],
    sample.fraction = hyper_grid$sample.fraction[i],
    verbose         = FALSE,
    seed            = 123,
    respect.unordered.factors = 'order',
  )
  # export OOB error 
  hyper_grid$rmse[i] <- sqrt(fit$prediction.error)
}

# hyperparameter grid

# 这是一个使用H2O平台进行随机森林回归的过程，其中参数解释如下：
# 
# hyper_grid: 超参数的网格，包含了要进行随机森林模型优化的参数列表。其中：
# 
# mtries: 随机选取的特征数量比例。
# min_rows: 每个叶子节点上最少需要的样本数量。
# max_depth: 每个树的最大深度。
# sample_rate: 每次训练时随机抽样的数据比例。
# search_criteria: 网格搜索策略参数，控制随机网格搜索的行为。其中：
# 
# strategy: 采用的策略，这里是随机离散型，即在超参数网格中随机选择一组参数。
# stopping_metric: 停止训练的度量标准，这里是均方误差（MSE）。
# stopping_tolerance: 停止训练的阈值，如果最近10次迭代中没有发生MSE的变化，就停止训练。
# stopping_rounds: 连续N个迭代中，如果没有MSE的变化，就停止训练。
# max_runtime_secs: 最长的训练时间。
# random_grid: 通过调用h2o.grid()函数，执行随机网格搜索，并传入以下参数：
# 
# algorithm: 要使用的算法，这里是随机森林。
# grid_id: 网格的名称。
# x: 自变量的名称。
# y: 因变量的名称。
# training_frame: 训练数据集的H2O对象。
# hyper_params: 超参数网格。
# ntrees: 树的数量。
# seed: 随机种子。
# stopping_metric: 停止训练的度量标准，这里是均方误差（MSE）。
# stopping_rounds: 连续N个迭代中，如果没有MSE的变化，就停止训练。
# stopping_tolerance: 停止训练的阈值，如果最近10次迭代中没有发生MSE的变化，就停止训练。
# search_criteria: 网格搜索策略参数。
# random_grid_perf: 获得网格搜索结果，并按照均方误差（MSE）排序。
hyper_grid <- list(
  mtries = floor(n_features * c(.05, .15, .25, .333, .4)),
  min_rows = c(1, 3, 5, 10),
  max_depth = c(10, 20, 30),
  sample_rate = c(.55, .632, .70, .80)
)

# re-run model with permutation-based variable importance
# 这段代码中首先使用了ranger函数生成一个随机森林模型，使用了permutation方法来计算变量的重要性，具体参数如下：
# 
# formula：因变量Sale_Price与自变量的公式
# data：训练数据集
# num.trees：生成的树的数量，这里为2000
# mtry：每棵树用于分裂节点的随机选择的变量数，这里为32
# min.node.size：叶节点中的最小观测数，这里为1
# sample.fraction：每次迭代使用的训练数据集中的观测比例，这里为0.8
# replace：每次训练时是否使用有放回的采样，这里为不使用，即FALSE
# importance：计算变量重要性的方法，这里使用permutation
# respect.unordered.factors：指示是否应在分类变量上应用顺序规则，这里使用order，即顺序规则
# verbose：是否打印详细的信息，这里为FALSE，即不打印
# seed：生成随机数的种子，这里为123
# 接着，使用了vip包中的vip函数来生成一个用于展示变量重要性的图表。其中，参数num_features指定了需要展示的变量数量，这里为25；参数bar指定了是否生成柱状图，这里为FALSE，即不生成。
rf_permutation <- ranger(
  formula = Sale_Price ~ ., 
  data = ames_train, 
  num.trees = 2000,
  mtry = 32,
  min.node.size = 1,
  sample.fraction = .80,
  replace = FALSE,
  importance = "permutation",
  respect.unordered.factors = "order",
  verbose = FALSE,
  seed  = 123
)

p1 <- vip::vip(rf_impurity, num_features = 25, bar = FALSE)
p2 <- vip::vip(rf_permutation, num_features = 25, bar = FALSE)
