# one knot

# 这行代码是利用 mda 包中的 mars() 函数来拟合一个 MARS 模型。具体解释如下：
# 
# mda::mars(): 调用 mda 包中的 mars() 函数，对数据进行 MARS 模型的训练。
# df$x: 指定自变量 x 作为 MARS 模型的输入。
# df$y: 指定因变量 y 作为 MARS 模型的输出。
# nk = 3: 指定 MARS 模型中最多可以有 3 个基函数。
# prune = FALSE: 指定不进行剪枝，即保留所有基函数。
# 
# 在广义加性模型（GAM）或者多项式回归中，基函数是指用来拟合函数形状的基本构建块。基函数可以是任何函数，通常是一些简单的函数，例如多项式、三角函数、指数函数等。在MARS（多元自适应回归样条）中，基函数是一些简单的分段函数，通常是以分段的线性函数或者分段的二次函数作为基础构建块，然后将这些基础构建块进行组合以构建更复杂的非线性函数形式。在MARS中，基函数的组合通过合并基函数来逐步构建。
# 在广义加性模型（GAM）或者多项式回归中，基函数是指用来拟合函数形状的基本构建块。基函数可以是任何函数，通常是一些简单的函数，例如多项式、三角函数、指数函数等。在MARS（多元自适应回归样条）中，基函数是一些简单的分段函数，通常是以分段的线性函数或者分段的二次函数作为基础构建块，然后将这些基础构建块进行组合以构建更复杂的非线性函数形式。在MARS中，基函数的组合通过合并基函数来逐步构建。
mars1 <- mda::mars(df$x, df$y, nk = 3, prune = FALSE)

# 这段代码是将数据框 df 中的 x 和 y 列作为横纵坐标绘制散点图，并在图上添加一条蓝色的线段。线段的纵坐标是由 MARS 模型 mars1 对 x 变量拟合得到的预测值，即 predicted 列。同时，图的标题为 “(A) One knot”。其中，“One knot” 是指 MARS 模型中使用的基函数是带一个节点的基函数。
p1 <- df %>%
  mutate(predicted = as.vector(mars1$fitted.values)) %>%
  ggplot(aes(x, y)) +
  geom_point(size = 1, alpha = .2) +
  geom_line(aes(y = predicted), size = 1, color = "blue") +
  ggtitle("(A) One knot")

# Fit a basic MARS model

# earth是一个R语言的包，提供了一种非参数的回归建模技术，称为MARS（Multivariate Adaptive Regression Splines，多元自适应回归样条）。MARS方法通过将预测变量空间分为多个小区域，用每个区域内变量的线性组合来逼近目标变量。MARS可以自动地处理线性和非线性关系，它可以自动选择重要的变量，并在模型中使用它们的相应交互项，还可以处理缺失值和离群值。MARS方法的优点是具有良好的解释性，它生成的模型是可解释的，因为它可以很容易地将变量的重要性和变量之间的关系可视化。
# Sale_Price 是模型的目标变量，表示房屋销售价格。
# . 表示所有其他列都是模型的自变量，也就是模型用来预测目标变量的特征变量。
# data = ames_train 表示数据集是 ames_train，即训练集数据。
mars1 <- earth(
  Sale_Price ~ .,  
  data = ames_train   
)

# 这段代码使用了R中的caret包的train()函数，对MARS回归进行交叉验证（cross-validation）并选择最优的超参数组合。具体参数说明如下：
# 
# x和y分别是自变量和因变量；
# method = "earth"表示采用earth()函数进行建模，即采用MARS算法；
# metric = "RMSE"表示评价指标为均方根误差（Root Mean Square Error）；
# trControl = trainControl(method = "cv", number = 10)表示采用10折交叉验证；
# tuneGrid = hyper_grid表示超参数的搜索范围为hyper_grid。

cv_mars <- train(
  x = subset(ames_train, select = -Sale_Price),
  y = ames_train$Sale_Price,
  method = "earth",
  metric = "RMSE",
  trControl = trainControl(method = "cv", number = 10),
  tuneGrid = hyper_grid
)

# 
# Attrition ~ .：指定目标变量为 Attrition，并将其余变量用于训练模型。
# data = churn_train：指定训练数据为 churn_train 数据框。
# method = "glm"：指定要拟合的模型为广义线性模型（GLM）。

# preProc = c("zv", "center", "scale")：指定在训练模型前要对数据进行的预处理步骤。其中 "zv" 表示删除所有值都相同的列，而 "center" 和 "scale" 表示对所有列进行中心化和标准化。
# trControl = trainControl(method = "cv", number = 10)：指定交叉验证方法为 10 折交叉验证。
glm_mod <- train(
  Attrition ~ ., 
  data = churn_train, 
  method = "glm",
  family = "binomial",
  preProc = c("zv", "center", "scale"),
  trControl = trainControl(method = "cv", number = 10)
  )

# train regularized logistic regression model
set.seed(123)
# Attrition ~ .：响应变量为 Attrition，预测变量为除了 Attrition 外的所有变量。
# data = churn_train：使用 churn_train 数据集来拟合模型。
# method = "glmnet"：使用 glmnet 方法进行拟合，这是一种广义线性模型的正则化方法，用于解决高维数据下的变量选择和模型复杂度问题。
# family = "binomial"：响应变量 Attrition 是二元分类变量，因此使用二项式模型族进行拟合。
# preProc = c("zv", "center", "scale")：在拟合模型之前，先进行了一些数据预处理，包括去除方差为零的变量、中心化和标准化。
# trControl = trainControl(method = "cv", number = 10)：使用交叉验证方法进行模型训练，其中 number = 10 表示使用 10 折交叉验证。
# tuneLength = 10：对于 glmnet 方法中的正则化参数进行 10 次调参。
penalized_mod <- train(
  Attrition ~ ., 
  data = churn_train, 
  method = "glmnet",
  family = "binomial",
  preProc = c("zv", "center", "scale"),
  trControl = trainControl(method = "cv", number = 10),
  tuneLength = 10
  )
