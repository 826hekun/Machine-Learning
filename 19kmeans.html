# Generate data
create_data <- function(sd) {
  data_frame(
    x1 = c(rnorm(100, sd = sd), rnorm(100, sd = sd) + 3),
    x2 = c(rnorm(100, sd = sd), rnorm(100, sd = sd) - 2)
  ) %>%
    mutate(`W(Ck)` = case_when(
      sd == 0.5  ~ "Best",
      sd == 0.75 ~ "Better",
      sd == 1   ~ "Good"
    ))

# Create mode function

# 这段代码定义了一个函数mode_fun，该函数用于计算一组数据的众数（mode）。
mode_fun <- function(x){  
  which.max(tabulate(x))
}
# data.frame(cluster = mnist_clustering$cluster, actual = mnist$train$labels): 这行代码创建了一个数据框，其中包含两列：cluster列包含K均值聚类的结果（聚类簇标签），actual列包含MNIST数据集训练集的实际标签。

# %>% group_by(cluster) %>% mutate(mode = mode_fun(actual)): 这行代码使用管道操作符%>%对数据框进行处理。首先，使用group_by(cluster)将数据按照cluster列进行分组。然后，使用mutate函数对每个分组计算实际标签的众数，并将结果存储在名为mode的新列中。这里调用了之前定义的mode_fun函数。
# 
# %>% ungroup(): 这行代码使用ungroup函数取消数据框的分组状态，确保后续操作应用于整个数据框而不是特定的分组。
# 
# %>% mutate_all(factor, levels = 0:9): 这行代码使用mutate_all函数将数据框中的所有列转换为因子类型。同时，levels = 0:9参数指定了因子的级别（levels），这里是数字0到9，对应于MNIST数据集中的标签类别。
# 
# 综合起来，这段代码创建了一个数据框mnist_comparison，其中包含了K均值聚类的结果、实际标签以及计算得到的每个聚类的众数。通过将实际标签与聚类结果进行比较，并计算每个聚类的众数，可以评估聚类结果与实际标签之间的一致性。最后，将数据框中的所有列转换为因子类型，并指定了因子的级别。
mnist_comparison <- data.frame(
  cluster = mnist_clustering$cluster,
  actual = mnist$train$labels
) %>%
  group_by(cluster) %>%
  mutate(mode = mode_fun(actual)) %>%
  ungroup() %>%
  mutate_all(factor, levels = 0:9)

# 当数据中存在缺失值（NA值）时，K均值聚类（kmeans）和PAM（Partitioning Around Medoids）聚类方法的处理方式略有不同。
# 
# 对于K均值聚类（kmeans）方法：
# 
# kmeans方法对于缺失值敏感，如果数据中存在缺失值，kmeans函数将无法处理，并会引发错误。因此，在应用kmeans之前，需要先对数据进行缺失值处理，例如删除包含缺失值的观测行或使用合适的缺失值填充方法。
# 对于PAM聚类方法：
# 
# PAM方法对于缺失值是相对鲁棒的，可以处理带有缺失值的数据。PAM算法是基于中心点的聚类方法，它使用观测对象之间的距离进行计算，并根据距离选择代表性的中心点。因此，在使用PAM进行聚类时，不需要显式处理缺失值，可以直接将带有缺失值的数据传递给PAM函数进行聚类分析。
# 综上所述，如果数据中存在缺失值，PAM聚类方法更适合使用，因为它可以在不对缺失值进行特殊处理的情况下进行聚类分析。而kmeans聚类方法对于缺失值是敏感的，需要在应用之前对缺失值进行处理。
