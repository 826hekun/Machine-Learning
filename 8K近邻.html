# Create blueprint
# 首先，使用 recipe 函数定义了一个数据处理的流程 blueprint，该流程用于将 churn_train 中的数据进行预处理，包括去除所有名义变量中无变异的特征，将 Satisfaction、WorkLifeBalance 和 JobInvolvement 三个整型变量转化为因子变量，然后将所有名义变量转化为虚拟变量，最后对所有数值型变量进行居中和缩放操作。
# 
# 然后，使用 trainControl 函数定义了一个交叉验证方法 cv，该方法使用的是重复 K 折交叉验证，其中 K=10，重复 5 次。同时设置了 classProbs = TRUE 表示需要输出分类的概率值，以用于计算 AUC 等度量指标，summaryFunction = twoClassSummary 表示使用二分类模型的评价指标。
blueprint <- recipe(Attrition ~ ., data = churn_train) %>%
  step_nzv(all_nominal()) %>%
  step_integer(contains("Satisfaction")) %>%
  step_integer(WorkLifeBalance) %>%
  step_integer(JobInvolvement) %>%
  step_dummy(all_nominal(), -all_outcomes(), one_hot = TRUE) %>%
  step_center(all_numeric(), -all_outcomes()) %>%
  step_scale(all_numeric(), -all_outcomes())

# Create a resampling method
cv <- trainControl(
  method = "repeatedcv", 
  number = 10, 
  repeats = 5,
  classProbs = TRUE,                 
  summaryFunction = twoClassSummary
)

# Create a hyperparameter grid search
# 首先使用 expand.grid() 函数生成一个包含不同参数组合的网格，其中 k 是要调节的超参数，使用 seq() 函数生成从 1 到训练集行数除以 3（向下取整）的等差数列，长度为 20，然后对这些值进行取整操作，作为超参数的备选值。这样生成的 hyper_grid 是一个数据框。
# 
# 然后使用 train() 函数对 KNN 模型进行训练，并在训练过程中进行超参数调优，具体参数含义如下：
# 
# blueprint：通过 recipes 包中的 recipe() 函数定义的数据预处理流程。
# data：训练集数据。
# method：使用 KNN 模型进行分类。
# trControl：交叉验证方法的参数，这里使用重复交叉验证（"repeatedcv"）进行模型训练，共进行 10 次交叉验证，重复 5 次。
# tuneGrid：超参数的备选值，即上一步生成的 hyper_grid。
# metric：模型评价指标，这里是 ROC 曲线下面积（Area Under ROC Curve）。
# 最终返回的 knn_grid 包含了模型在不同超参数下的性能表现，可以通过 summary() 函数查看不同超参数下的评估指标（如 ROC-AUC、准确率、召回率等）的平均值和标准差
hyper_grid <- expand.grid(
  k = floor(seq(1, nrow(churn_train)/3, length.out = 20))
)

# Fit knn model and perform grid search
knn_grid <- train(
  blueprint, 
  data = churn_train, 
  method = "knn", 
  trControl = cv, 
  tuneGrid = hyper_grid,
  metric = "ROC"
)

ggplot(knn_grid)

# confusionMatrix: 该函数是用来计算分类模型的混淆矩阵，即模型预测结果与真实结果的对比，输出包括各种评价指标如准确率、召回率等等。在这里，该函数接收了两个参数，分别是KNN模型预测出的标签knn_mnist$pred$pred以及真实的标签knn_mnist$pred$obs。
# cm$byClass[, c(1:2, 11)]: 对于混淆矩阵cm，该行代码使用了byClass函数将敏感度（sensitivity）、特异度（specificity）和准确率（accuracy）提取出来进行输出。
# varImp(knn_mnist): varImp函数用于计算KNN模型的变量重要性（Variable Importance），返回各个特征的重要性得分，即在建立模型中，各个特征在预测目标上的重要性评估。

# Get median value for feature importance
# imp: 模型特征的重要性值。
# vi: 储存了使用KNN算法拟合MNIST数据集后，每个特征的重要性评分。
# rowname_to_column: 将行名称转换为列，以便进一步处理数据。
# gather: 将数据从宽格式变为长格式，以便进一步处理数据。
# group_by: 按特征名称对数据进行分组。
# summarize: 对每个特征的重要性进行求中位数的操作。
# edges: 存储所有边缘像素的信息，包括特征名称和重要性评分。其中，nzv 存储了所有方差为0的特征的索引，这些特征对模型的贡献较小，因此将它们的重要性评分设为0。
imp <- vi$importance %>%
  rownames_to_column(var = "feature") %>%
  gather(response, imp, -feature) %>%
  group_by(feature) %>%
  summarize(imp = median(imp))

# Create tibble for all edge pixels
edges <- tibble(
  feature = paste0("V", nzv),
  imp = 0
)

# Combine and plot
# imp: 包含所有像素重要性信息的数据框。
# edges: 包含未被删除的边缘像素信息的数据框。
# rbind(imp, edges): 将 imp 和 edges 数据框按行合并，用于包含所有像素的信息。
# as.numeric(str_extract(feature, "\\d+")): 从列名 feature 中提取数字，并将其转换为数值类型。
# arrange(ID): 根据像素的ID（即像素在图像中的位置）对数据框进行排序。
# image(matrix(imp$imp, 28, 28), col = gray(seq(0, 1, 0.05)), xaxt="n", yaxt="n"): 将重要性信息可视化为图像，其中像素的重要性值被映射到灰度值。函数中的 matrix() 函数将一维向量转换为矩阵，col 参数设置了灰度值的分布，xaxt 和 yaxt 参数用于禁用坐标轴刻度。
imp <- rbind(imp, edges) %>%
  mutate(ID  = as.numeric(str_extract(feature, "\\d+"))) %>%
  arrange(ID)
image(matrix(imp$imp, 28, 28), col = gray(seq(0, 1, 0.05)), 
      xaxt="n", yaxt="n")
